async function showBooks(){

  state.view="books";

  $("app").innerHTML = `<div class="sec-head"><h2>書籍目錄</h2><span class="count" id="bkCount"></span><span class="grow"></span>

function newBook(){ editingBookId=null; $("bookModalTitle").textContent="新增書籍"; $("bkTitle").value=""; $("bkAuthor").value=""; $("bkDesc").value=""; openModal("bookModal"); }

async function saveBook(){

  const title=$("bkTitle").value.trim();

  if(!title){ toast("請填寫書名"); return; }

  const row={ title, author:$("bkAuthor").value.trim(), description:$("bkDesc").value.trim() };

  const q = editingBookId ? db.from("books").update(row).eq("id",editingBookId) : db.from("books").insert(row);

  const { error } = await q;

  if(error){ toast("儲存失敗："+error.message); return; }

  closeModal("bookModal"); toast("已儲存"); render();

}

async function deleteBook(){

  const { error } = await db.from("books").delete().eq("id",state.bookId);

  if(error){ toast("刪除失敗："+error.message); return; }

  toast("已刪除"); showBooks();

}

  $("app").innerHTML = `

    <div class="book-hero">

      <h2>${esc(book.title)}</h2>

      ${book.author?`<div class="ba">作者：${esc(book.author)}</div>`:""}

      ${book.description?`<div class="bd">${esc(book.description)}</div>`:""}

    </div>

    <div class="sec-head"><h2>章節內容</h2><span class="count">共 ${chapters.length} 章</span><span class="grow"></span>

      <button class="btn navy admin-only" onclick="newChapter()">＋ 新增章節</button>

      <button class="btn navy admin-only" onclick="editBook()">編輯書籍</button>

    <div class="ch-list">${ chapters.length ? chapters.map(c=>`

}


function newChapter(){ $("chOrder").value=""; $("chTitle").value=""; $("chFile").value=""; openModal("chModal"); }

async function saveChapter(){

  if(file.type!=="application/pdf"){ toast("請選擇 PDF 檔案"); return; }

  $("chSaveBtn").disabled=true; $("chProgress").style.display="block";

  const path = `${state.bookId}/${Date.now()}.pdf`;

  const { error: upErr } = await db.storage.from(CONFIG.BUCKET).upload(path, file, { contentType:"application/pdf" });

  if(upErr){ toast("PDF 上傳失敗："+upErr.message); $("chSaveBtn").disabled=false; $("chProgress").style.display="none"; return; }

  const { error } = await db.from("chapters").insert({ book_id:state.bookId, sort_order:order, title, pdf_path:path });

  $("chSaveBtn").disabled=false; $("chProgress").style.display="none";

  if(error){ toast("儲存失敗："+error.message); return; }

  closeModal("chModal"); toast("章節已建立"); openBook(state.bookId);

}

async function deleteChapter(id, pdfPath){

  if(!confirm("確定刪除此章節？相關心得會一併刪除。")) return;

  await db.storage.from(CONFIG.BUCKET).remove([pdfPath]);

  const { error } = await db.from("chapters").delete().eq("id",id);

  if(error){ toast("刪除失敗："+error.message); return; }

  toast("已刪除"); openBook(state.bookId);

}


/* ---------- 章節頁（PDF + 心得研討） ---------- */

function pdfUrl(path){ return db.storage.from(CONFIG.BUCKET).getPublicUrl(path).data.publicUrl; }


async function openChapter(id){

  state.view="chapter"; state.chapterId=id;

  const { data: ch } = await db.from("chapters").select("*, books(id,title)").eq("id",id).single();

  if(!ch){ showBooks(); return; }

  state.chapter=ch; state.bookId=ch.books.id; state.book={id:ch.books.id,title:ch.books.title};

  const url = pdfUrl(ch.pdf_path);

  $("app").innerHTML = `

    <div class="crumb"><a onclick="showBooks()">書籍目錄</a><span class="sep">›</span><a onclick="openBook('${ch.books.id}')">${esc(ch.books.title)}</a><span class="sep">›</span>${esc(ch.title)}</div>

    <div class="ch-head"><div class="eyebrow">CHAPTER ${String(ch.sort_order).padStart(2,"0")}</div><h2>${esc(ch.title)}</h2></div>

    <div class="pdf-bar">

      <button class="btn navy" onclick="window.open('${url}','_blank')">在新視窗開啟 PDF</button>

      <a href="${url}" download style="font-size:13.5px">下載 PDF</a>

    </div>

    <embed class="pdf-frame" src="${url}" type="application/pdf">

    <div class="notice">本章節內容僅供內部研讀，請勿外流。手機若無法直接預覽，請點「在新視窗開啟 PDF」。</div>


    <div class="disc">

      <div class="disc-head"><span class="dot"></span><h3>學習心得研討</h3><span class="count" id="postCount"></span></div>

      <div class="post-form">

        <div class="row">

          <div><label>您的姓名 *</label><input type="text" id="pName" placeholder="請填真實姓名，方便同仁交流" value="${esc(localStorage.getItem("rc_name")||"")}"></div>

        </div>

        <textarea id="pContent" placeholder="讀完這個章節，您的收穫、反思、或想跟大家討論的問題⋯"></textarea>

        <div class="form-foot"><button class="btn solid" onclick="submitPost(null)">發布心得</button></div>

      </div>

      <div id="postList">載入中⋯</div>

    </div>`;

  loadPosts();

}


async function loadPosts(){

  const { data: posts, error } = await db.from("posts").select("*").eq("chapter_id",state.chapterId).order("created_at",{ascending:true});

  if(error){ $("postList").innerHTML=`<div class="empty">心得載入失敗：${esc(error.message)}</div>`; return; }

  const tops = posts.filter(p=>!p.parent_id);

  const replies = pid => posts.filter(p=>p.parent_id===pid);

  $("postCount").textContent = `共 ${posts.length} 則`;

  if(!tops.length){ $("postList").innerHTML=`<div class="empty">還沒有人分享心得，成為第一位分享者吧。</div>`; return; }

  $("postList").innerHTML = tops.map(p=>renderPost(p,false)+replies(p.id).map(r=>renderPost(r,true)).join("")+`<div id="rf-${p.id}"></div>`).join("");

}


function renderPost(p, isReply){

  return `<div class="post ${isReply?"reply":""}">

    <div class="p-head"><span class="p-name">${esc(p.author_name)}</span><span class="p-time">${fmtTime(p.created_at)}</span></div>

    <div class="p-body">${esc(p.content)}</div>

    <div class="p-actions">

      ${!isReply?`<button onclick="showReplyForm('${p.id}')">回覆</button>`:""}

      ${isAdmin()?`<button class="del" onclick="deletePost('${p.id}')">刪除</button>`:""}

    </div></div>`;

}


function showReplyForm(pid){

  const slot=$("rf-"+pid);

  if(slot.innerHTML){ slot.innerHTML=""; return; }

  slot.innerHTML=`<div class="post-form reply-form">

    <div class="row"><div><label>您的姓名 *</label><input type="text" id="rName-${pid}" value="${esc(localStorage.getItem("rc_name")||"")}"></div></div>

    <textarea id="rContent-${pid}" placeholder="回應這則心得⋯"></textarea>

    <div class="form-foot"><button class="btn plain" onclick="$('rf-${pid}').innerHTML=''">取消</button>

    <button class="btn solid" onclick="submitReply('${pid}')">發布回覆</button></div></div>`;

}


async function submitPost(parentId){

  const name=$("pName").value.trim(), content=$("pContent").value.trim();

  if(!name||!content){ toast("姓名與內容皆為必填"); return; }

  localStorage.setItem("rc_name",name);

  const { error } = await db.from("posts").insert({ chapter_id:state.chapterId, parent_id:parentId, author_name:name, content });

  if(error){ toast("發布失敗："+error.message); return; }

  $("pContent").value=""; toast("已發布"); loadPosts();

}

async function submitReply(pid){

  const name=$("rName-"+pid).value.trim(), content=$("rContent-"+pid).value.trim();

  if(!name||!content){ toast("姓名與內容皆為必填"); return; }

  localStorage.setItem("rc_name",name);

  const { error } = await db.from("posts").insert({ chapter_id:state.chapterId, parent_id:pid, author_name:name, content });

  if(error){ toast("發布失敗："+error.message); return; }

  toast("已發布"); loadPosts();

}

async function deletePost(id){

  if(!confirm("確定刪除這則心得（含其回覆）？")) return;

  const { error } = await db.from("posts").delete().eq("id",id);

  if(error){ toast("刪除失敗："+error.message); return; }

  toast("已刪除"); loadPosts();

}

async function exportAll(){

  toast("正在整理備份資料⋯");

  const [{data:books},{data:chapters},{data:posts}] = await Promise.all([

    db.from("books").select("*").order("created_at"),

    db.from("chapters").select("*").order("book_id"),

    db.from("posts").select("*").order("created_at")

  ]);

  const backup = {

    exported_at: new Date().toISOString(),

    note: "GOTREND 讀書研討會平台完整備份。pdf_url 為各章節 PDF 的下載連結，可逐一下載保存。",

    books, 

    chapters: chapters.map(c=>({ ...c, pdf_url: pdfUrl(c.pdf_path) })),

    posts

  };

  const blob = new Blob([JSON.stringify(backup,null,2)],{type:"application/json"});

  const a=document.createElement("a");

  a.href=URL.createObjectURL(blob);

  a.download=`讀書會平台備份_${new Date().toISOString().slice(0,10)}.json`;

  a.click();

  toast("備份檔已下載（含所有 PDF 下載連結）");

}
